CREATE OR REPLACE PROCEDURE GETNCBYID(ID IN NUMBER)
	IS
		CATA NEWCATEGORY.TITLE%TYPE;
		DESCRIPT NEWCATEGORY.DESCRIPTION%TYPE;
	BEGIN
		SELECT TITLE, DESCRIPTION INTO CATA, DESCRIPT
		FROM NEWCATEGORY
		WHERE CATID=ID;
		DBMS_OUTPUT.PUT_LINE(CATA || '...'|| DESCRIPT);
END;

CREATE OR REPLACE PROCEDURE COPY()
IS
			CURSOR ALL_RECORDS SELECT * FROM QA;
			R QA.REGNO%TYPE;
			N QA.NAME%TYPE;
			G QA.GRP%TYPE;
			A QA.AREA%TYPE;
			M QA.MARKS%TYPE;
Begin
		OPEN ALL_RECORDS;
		Loop
		FETCH ALL_RECORDS INTO R,N,G,A,M;
		IF ALL_RECORDS%NOTFOUND THEN
			EXIT;
		END IF;
		IF G='ONS' THEN
			INSERT INTO BACKUPONS VALUES(R,N,G,A,M);
		END IF;
		IF G='QAC' THEN
			INSERT INTO BACKUPQAC VALUES(R,N,G,A,M);
		END IF;
	END LOOP;
	CLOSE ALL_RECORDS;
END;
